\subsection{How to integrate Kokkos}

Note the difference in the version number between \texttt{x.y.z} and
\texttt{x.y.zz}.

\subsubsection{As an external dependency}

\paragraph{Configure, build and install Kokkos}

\begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{git}\NormalTok{ clone {-}b x.y.zz https://github.com/kokkos/kokkos.git}
\BuiltInTok{cd}\NormalTok{ kokkos}
\FunctionTok{cmake}\NormalTok{ {-}B build }\KeywordTok{\textbackslash{}}
    \ExtensionTok{{-}DCMAKE\_CXX\_COMPILER}\NormalTok{=}\OperatorTok{\textless{}}\NormalTok{your C++ compiler}\OperatorTok{\textgreater{}} \KeywordTok{\textbackslash{}}
    \ExtensionTok{{-}DCMAKE\_INSTALL\_PREFIX}\NormalTok{=path/to/kokkos/install }\KeywordTok{\textbackslash{}}
    \OperatorTok{\textless{}}\ExtensionTok{Kokkos}\NormalTok{ compile options}\OperatorTok{\textgreater{}}
\FunctionTok{cmake}\NormalTok{ {-}{-}build build}
\FunctionTok{cmake}\NormalTok{ {-}{-}install build}
\end{Highlighting}
\end{Shaded}

\paragraph{Setup, and configure your code}

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{find\_package}\NormalTok{(Kokkos x.y.z }\OtherTok{REQUIRED}\NormalTok{)}
\KeywordTok{target\_link\_libraries}\NormalTok{(}
\NormalTok{    my{-}app}
\NormalTok{    Kokkos::kokkos}
\NormalTok{)}
\end{Highlighting}
\end{Shaded}

\begin{Shaded}
\begin{Highlighting}[]
\BuiltInTok{cd}\NormalTok{ path/to/your/code}
\FunctionTok{cmake}\NormalTok{ {-}B build }\KeywordTok{\textbackslash{}}
    \ExtensionTok{{-}DCMAKE\_CXX\_COMPILER}\NormalTok{=}\OperatorTok{\textless{}}\NormalTok{your C++ compiler}\OperatorTok{\textgreater{}} \KeywordTok{\textbackslash{}}
    \ExtensionTok{{-}DKokkos\_ROOT}\NormalTok{=path/to/kokkos/install}
