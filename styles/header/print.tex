\fancypagestyle{kokkos}{%
    \setlength{\headerheight}{1cm}
    \setlength{\footerheight}{0.5cm + 0.25 \printmargin}
    \setlength{\horizontalskip}{0.25cm}

    \newgeometry{
        hmargin=0.5cm,
        vmargin=0cm,
        includeheadfoot,
        headheight=\headerheight,
        headsep=0.5cm,
        footskip=\footerheight + 2\horizontalskip,
    }
    \renewcommand{\headrulewidth}{0pt}
    \renewcommand{\footrulewidth}{0pt}

    \fancyhf{}

    \fancyhead{%
        % Add the title and the page and the version
        \setlength{\fboxsep}{0pt}
        \colorbox{main}{%
            \parbox[b][\headerheight][c]{\headwidth}{%
                \fancycenter{%
                    \hspace{\horizontalskip}%
                    {\LARGE\bfseries\href{https://kokkos.org/kokkos-core-wiki}{\gettitle}}%
                    {\large\bfseries\color{lightmain}\ :: page \thepage}%
                }{}{%
                    \small\color{lightmain}v\input{VERSION}%
                    \hspace{\horizontalskip}
                }
            }%
        }%
    }%

    \fancyfoot{%
        % Add the Kokkos wiki link
        \setlength{\fboxsep}{0pt}
        \colorbox{main}{%
            \begin{minipage}[b][\footerheight][t]{\headwidth}
                \vspace{0.2em}%
                \ifthenelse{\isodd{\value{page}}}{%
                }{%
                    \hspace{\printmargin}%
                }%
                \hspace{\horizontalskip}%
                \footnotesize\bfseries\color{lightmain}\strut\url{https://kokkos.org/kokkos-core-wiki}%
            \end{minipage}%
        }%
    }%
}

        hmargin=0.5cm,
        vmargin=0cm,
        includeheadfoot,
        headheight=\headerheight,
        headsep=0.5cm,
        footskip=\footerheight + 2\horizontalskip,
    }

    \fancyhead{%
        % Add the title and the page
        \setlength{\fboxsep}{0pt}
        \colorbox{main}{%
            \parbox[b][\headerheight][b]{\headerheight}{%
                % add left continuation of the logo
                \ifthenelse{\isodd{\value{page}}}{%
                }{%
                    \colorbox{lightmain}{%
                        \parbox[b][\headerheight]{\printmargin}{~}%
                    }%
                }%
                \begin{minipage}[b][\headerheight][b]{\headerheight - \printmargin}
                    \setlength{\logomaxheight}{\headerheight - 0.75\printmargin}
                    % add top continuation of the logo
                    \colorbox{lightmain}{%
                        \parbox[b][\printmargin][b]{
                            0.575\logomaxheight % manual adjust logo height
                        }{~}%
                    }%
                    \vspace{-0.04cm} % manual adjust vertical gap
                    \\
                    \includegraphics[
                        height=\logomaxheight,
                    ]{images/kokkos_wire.pdf}%
                \end{minipage}
            }%
            \hspace{-\headerheight}
            \parbox[b][\headerheight][c]{\headwidth}{%
                \fancycenter{%
                    \ifthenelse{\isodd{\value{page}}}{%
                    }{%
                        \hspace{\printmargin}%
                    }%
                    \hspace{0.4\headerheight}
                    {\Huge\bfseries{}\href{https://kokkos.org/kokkos-core-wiki}{\gettitle}}%
                    {\Large\bfseries\color{lightmain}\ :: page \thepage}%
                }{}{%
                    \small\color{lightmain}v\input{VERSION}%
                    \hspace{\horizontalskip}
                    \ifthenelse{\isodd{\value{page}}}{%
                        \hspace{\printmargin}%
                    }{}%
                }
            }%
        }%
    }%

% fill with dotted lines until the end of the page
\newcommand{\dotcolumnfill}[1]{%
    % https://tex.stackexchange.com/a/319436
    \par
    \null
    \vskip -\ht\strutbox
    \xleaders \hb@xt@ \hsize {%
        \strut \cleaders \hb@xt@ .44em{\hss.\hss}\hfill
    }\vfill
    #1
    \vskip \ht\strutbox
    \break
}

\makeatother
